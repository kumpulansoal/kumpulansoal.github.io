f(x) = kx + l

f(4) = 4k + l
8 = 4k + l ...(1)

f(-1) = -k + l
18 = -k + l
k = l - 18 ...(2)

Substitusi (2) ke (1)
8 = 4(l - 18) + l
8 = 4l - 72 + l
80 = 5l
l = 16

substitusi l = 16 ke (2)
k = 16 - 18
k = -2

maka, rumus fungsi
f(x) = -2x + 16
